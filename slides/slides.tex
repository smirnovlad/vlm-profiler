\section{Goal and setup}
%----------------------------------------------------------------------------------------

\begin{frame}
	\frametitle{The question}

	\textbf{Project goal:} understand how VLMs scale and where to spend optimization effort.

	\bigskip

	Two concrete questions:
	\begin{enumerate}
		\item Which models are Pareto-optimal on quality / latency / energy?
		\item \alert{Which parts of the model dominate cost?} (the supervisor's main question)
	\end{enumerate}

	\bigskip

	\textbf{Approach:} vary one axis at a time, measure four things directly, then split the dominant model into sub-components.
\end{frame}

%------------------------------------------------

\begin{frame}
	\frametitle{Setup}
	\framesubtitle{7 models, 3 datasets, 5 axes, 4 metrics}

	\footnotesize
	\begin{columns}[t]
		\begin{column}{0.55\textwidth}
			\textbf{Models (7)}
			\begin{itemize}\setlength\itemsep{0pt}
				\item \texttt{blip2-opt-2.7b}, \texttt{blip2-flan-t5-xl}
				\item \texttt{instructblip-flan-t5-xl}
				\item \texttt{instructblip-vicuna-7b}
				\item \texttt{llava-1.5-7b}, \texttt{llava-1.5-13b}
				\item \texttt{fuyu-8b}
			\end{itemize}
			\smallskip
			\textbf{Datasets} (300 samples each):\\
			ScienceQA, TextVQA, COCO Captions
		\end{column}
		\begin{column}{0.45\textwidth}
			\textbf{Axes varied}
			\begin{itemize}\setlength\itemsep{0pt}
				\item Resolution: 224 / 336 / 448 / 512
				\item Prompt: 10 / 50 / 100 / 200 tokens
				\item Batch: 1 / 2 / 4 / 8
				\item FP32, FP16, \texttt{torch.compile}, FA2
				\item GPU vs.\ CPU
			\end{itemize}
			\smallskip
			\textbf{Metrics}: latency, FLOPs, energy (J), WER.\\
			\textbf{Hardware}: 2× RTX 5880 Ada, 48 GiB.
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}

	\begin{columns}[t]
		\begin{column}{0.5\textwidth}
			\footnotesize
			\textbf{Resolution 224 → 512:} flat for everyone except Fuyu ($+15\%$ at 512 — raw image patches as text tokens).
			\smallskip

			\textbf{Prompt 10 → 200:} T5 models slow $2{-}3.5\times$; 7B+ decoder-only models barely move.
			\smallskip

			\textbf{Batch 1 → 8:} small models gain throughput near-linearly; 7B+ scale linearly in \emph{latency} — no throughput gain.
			\smallskip

			\textbf{CPU vs GPU:} 15–30× gap. Impractical for 7B+.
		\end{column}
		\begin{column}{0.5\textwidth}
			\includegraphics[width=\textwidth]{optimization_speedup.png}\\[1mm]
			\tiny FP16 helps OPT/Vicuna/LLaMA backbones (1.85–2.2×). \emph{Slows} T5. \texttt{torch.compile}: no measurable speedup.
		\end{column}
	\end{columns}
\end{frame}

		\texttt{torch.compile} on T5 &
			Compiler bails on encoder-decoder graph. No measurable speedup on the others either. \\
		FP16 on T5 &
			\emph{Slower} than FP32 — internal dtype casts in T5's forward. \\
		FP16 on \texttt{fuyu-8b} &
			Float / Half dtype mismatch in patch embedding. \\
		\texttt{flash-attn} pip & Build needs \texttt{nvcc} $\geq$ 11.7; system has 11.5. Used PyTorch SDPA, which dispatches FA2 in FP16 on Ada. \\
		Batched \texttt{fuyu-8b} & Processor returns lists for \texttt{batch>1}. Skipped batch sweep for Fuyu. \\
		\bottomrule
	\end{tabular}
\end{frame}

\section{Where the time goes}
%----------------------------------------------------------------------------------------

\begin{frame}
	\frametitle{The punchline: where the time goes}

	\begin{columns}[t]
		\begin{column}{0.62\textwidth}
			\includegraphics[width=\textwidth]{component_latency_share.png}
		\end{column}
		\begin{column}{0.38\textwidth}
			\scriptsize Forward hooks + CUDA events on top-level submodules.
			\smallskip

			\textbf{LLM decoder share}
			\begin{itemize}\setlength\itemsep{0pt}
				\item BLIP-2 OPT 2.7b: \textbf{74\%}
				\item BLIP-2 / InstructBLIP T5: 84–87\%
				\item 7B+ models: \textbf{98–99.99\%}
			\end{itemize}

			\smallskip
			\alert{\textbf{Optimize the decoder.}} Vision encoder is \emph{not} the bottleneck — explains flat latency-vs-resolution.
		\end{column}
	\end{columns}
\end{frame}

%------------------------------------------------

\begin{frame}
	\frametitle{Inside the decoder: prefill vs.\ per-token decode}

	\begin{columns}[t]
		\begin{column}{0.6\textwidth}
			\includegraphics[width=0.98\textwidth]{prefill_decode_crossover.png}
		\end{column}
		\begin{column}{0.4\textwidth}
			\scriptsize
			$t(N) = t_{\text{prefill}} + N\,t_{\text{decode}}$, fit at $N{=}1, 50$.

			\textbf{Crossover $N^*$:}
			\begin{itemize}\setlength\itemsep{0pt}
				\item fuyu-8b 1.6, vicuna-7b 4.1
				\item llava-13b 5.1, llava-7b 15
				\item BLIP-2 family 44–54
			\end{itemize}

			\textbf{Lever:} short ($\leq 10$ tok) → prefill (FA2). Long ($\geq 50$) → decode (spec., KV quant).
		\end{column}
	\end{columns}
\end{frame}

%------------------------------------------------

\begin{frame}
	\frametitle{FlashAttention-2: smaller win than expected}

	\footnotesize
	\begin{table}
		\centering
		\begin{tabular}{@{}lcccc@{}}
			\toprule
			 & SDPA prefill & eager prefill & FA2 speedup & decode/tok (SDPA / eager) \\
			\midrule
			\texttt{llava-1.5-7b}  &  94.5 ms & 105.4 ms & \textbf{1.11×} & 15.3 / 15.6 ms \\
			\texttt{llava-1.5-13b} & 169.0 ms & 181.3 ms & \textbf{1.07×} & 33.0 / 34.2 ms \\
			\bottomrule
		\end{tabular}
	\end{table}

	\smallskip

	\begin{block}{Why so modest?}
		\footnotesize At plen=10 the LLaVA input is $\approx 586$ tokens (10 text + 576 image patch). The eager $O(N^2)$ attention matrix still fits in cache at $N\approx 600$.
	\end{block}

	\begin{exampleblock}{The real lever was the dtype flip}
		\footnotesize FP32→FP16 on \texttt{llava-7b}: 4.6× prefill, 1.9× decode. FA2 inside FP16 adds 1.07–1.11×. For short-answer VQA, FA2 is not the lever — but it would scale at long context.
	\end{exampleblock}
\end{frame}

\section{Insights and next steps}
%----------------------------------------------------------------------------------------

\begin{frame}
	\frametitle{Practical insights}

	\footnotesize
	\begin{enumerate}\setlength\itemsep{2pt}
		\item \textbf{For deployment today:} \texttt{blip2-flan-t5-xl} or \texttt{blip2-opt-2.7b}. The 7B+ tier's quality gain does not justify $\sim$15× cost.
		\item \textbf{Optimize the decoder, not the vision tower}, on every model $\geq$ 7B.
		\item \textbf{Match the optimization to generation length} via $N^*$: short answers → prefill-side (FA2, chunked prefill); long answers → decode-side (speculative, INT8/INT4, KV-cache quant).
		\item \textbf{FP16 first, attention kernels second.} Dtype flip is a 2–4× lever; FA2 inside FP16 is 1.07–1.11× at short context.
		\item \textbf{T5 is a trap with current tooling}: \texttt{torch.compile} fails, FP16 makes it slower.
		\item \textbf{Fuyu's design choice has a cost}: tokenized image patches inflate KV cache → 41 ms/token despite being 8B.
	\end{enumerate}
\end{frame}

%------------------------------------------------

\begin{frame}
	\frametitle{Limitations and what's next}

	\footnotesize
	\begin{columns}[t]
		\begin{column}{0.5\textwidth}
			\textbf{Limitations}
			\begin{itemize}\setlength\itemsep{0pt}
				\item FLOPs for T5 backbones estimated as $2 \times \text{params}$; calflops doesn't trace cleanly.
				\item Energy from \texttt{nvidia-smi} at 100 ms polling — a few percent precision, not lab-grade.
				\item No quantization (INT8/INT4) measured.
				\item FA2 evaluated only at short context; benefit at $N \gg 1k$ not captured.
			\end{itemize}
		\end{column}
		\begin{column}{0.5\textwidth}
			\textbf{Next steps}
			\begin{itemize}\setlength\itemsep{0pt}
				\item INT8 / GPTQ / AWQ on 7B+ tier (decode-bound, biggest win).
				\item Speculative decoding with a small draft model.
				\item Long-context FA2 sweep (resolution 1024+, multi-image prompts).
				\item Fix the \texttt{idefics2-8b} integration.
			\end{itemize}
		\end{column}
	\end{columns}
\end{frame}
